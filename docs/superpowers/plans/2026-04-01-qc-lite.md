# QC Lite Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add quality control analysis to LUCID — reprojection/epipolar histograms with draggable thresholds, node/instance swap detection, jitter detection, limb length consistency, worst-first issue list with frame navigation, and a downsample option for large datasets.

**Architecture:** Port `qc.js` from `eric/qc-control` branch, update all `trackIdx` references to `identityId` for 3D standardization compatibility, trim removed features (composite score, bodypart-camera table), add QC tab to info panel with histograms and issue list, add toolbar buttons and keyboard shortcuts.

**Tech Stack:** Vanilla JS, Canvas 2D (histograms), existing triangulation engine

```javascript
        // ============================================
        // Quality Control
        // ============================================

        function qcNextIssue() {
            if (!state.qcResults || state.qcResults.flaggedFrames.size === 0) return;
            var next = QC.nextFlaggedFrame(state.qcResults.flaggedFrames, state.currentFrame);
            if (next != null && videoController) {
                videoController.seekToFrame(next);
            }
        }

        function qcPrevIssue() {
            if (!state.qcResults || state.qcResults.flaggedFrames.size === 0) return;
            var prev = QC.prevFlaggedFrame(state.qcResults.flaggedFrames, state.currentFrame);
            if (prev != null && videoController) {
                videoController.seekToFrame(prev);
            }
        }

        async function runQCAnalysis() {
            if (!state.session) { setStatus('No session', 'warning'); return; }
            if (!sessionHasCalibration()) { showCalibrationRequiredPopup(); return; }

            var sampleRate = parseInt(document.getElementById('qcSampleRate').value) || 0;

            // Determine which frames to triangulate
            var allFrameIndices = [];
            for (var [fIdx] of state.session.instanceGroups) {
                allFrameIndices.push(fIdx);
            }
            allFrameIndices.sort(function (a, b) { return a - b; });

            var frameIndices;
            if (sampleRate > 0) {
                frameIndices = [];
                for (var si = 0; si < allFrameIndices.length; si += sampleRate) {
                    frameIndices.push(allFrameIndices[si]);
                }
            } else {
                frameIndices = allFrameIndices;
            }

            // Triangulate frames that aren't already done
            showLoading('Triangulating ' + frameIndices.length + ' frames...');
            var triCount = 0;
            var cameras = state.session.cameras;
            for (var fi = 0; fi < frameIndices.length; fi++) {
                var frameIdx = frameIndices[fi];
                if (state.triangulationResults.has(frameIdx)) { triCount++; continue; }

                var frameGroupsList = state.session.instanceGroups.get(frameIdx);
                if (!frameGroupsList || frameGroupsList.length === 0) continue;

                var frameResults = [];
                for (var gi = 0; gi < frameGroupsList.length; gi++) {
                    var group = frameGroupsList[gi];
                    var viewsWithLabels = 0;
                    for (var cj = 0; cj < cameras.length; cj++) {
                        var inst = group.getInstance(cameras[cj].name);
                        if (inst && inst.points && inst.points.some(function (p) { return p != null; })) {
                            viewsWithLabels++;
                        }
                    }
                    if (viewsWithLabels < 2) continue;

                    var groupCameras = cameras.filter(function (c) { return group.cameraNames.indexOf(c.name) >= 0; });
                    var result = triangulateAndReproject(group, groupCameras);
                    group.points3d = result.points3d;
                    group.reprojections = result.reprojections;
                    storeReprojectedInstances(group, result, cameras);
                    group.markClean();

                    frameResults.push({
                        group: group,
                        points3d: result.points3d,
                        reprojections: result.reprojections,
                        errors: result.errors,
                        meanError: result.meanError,
                    });
                }
                if (frameResults.length > 0) {
                    state.triangulationResults.set(frameIdx, frameResults);
                    triCount++;
                }

                if (fi > 0 && fi % 100 === 0) {
                    showLoading('Triangulating... ' + fi + '/' + frameIndices.length);
                    await new Promise(function (r) { setTimeout(r, 0); });
                }
            }

            // Run QC analysis
            showLoading('Running QC analysis on ' + triCount + ' frames...');
            await new Promise(function (r) { setTimeout(r, 10); });

            try {
                state.qcResults = await QC.runFullAnalysis(
                    state.session,
                    state.triangulationResults,
                    null,
                    function (msg) { showLoading(msg); }
                );
            } catch (e) {
                console.error('[QC] Analysis failed:', e);
                hideLoading();
                setStatus('QC analysis failed: ' + e.message, 'error');
                return;
            }

            hideLoading();
            displayQCResults();
            setStatus('QC: ' + state.qcResults.flaggedFrames.size + ' flagged frames', 'success');
        }

        function displayQCResults() {
            if (!state.qcResults) return;
            var r = state.qcResults;

            // Show stats
            document.getElementById('qcNotRun').style.display = 'none';
            document.getElementById('qcStats').style.display = 'block';
            document.getElementById('qcStatsLine').textContent =
                r.flaggedFrames.size + ' flagged frames, ' +
                r.sortedIssues.length + ' issues';

            // Show sections
            document.getElementById('qcHistogramSection').style.display = '';
            document.getElementById('qcIssueListSection').style.display = '';

            // Draw histograms
            var thresh = r.autoThresholds;
            var dist = r.distributions;
            QC.drawHistogram(document.getElementById('qcHistReproj'),
                dist.reproj, thresh.reproj, { title: 'Reprojection (px)', color: '#6b7280' });
            QC.drawHistogram(document.getElementById('qcHistEpipolar'),
                dist.epipolar, thresh.epipolar, { title: 'Epipolar (px)', color: '#6b7280' });

            // Toolbar badge
            var badge = document.getElementById('qcToolbarBadge');
            if (r.flaggedFrames.size > 0) {
                badge.textContent = r.flaggedFrames.size;
                badge.style.display = '';
            } else {
                badge.style.display = 'none';
            }

            // Issue list
            renderQCIssueList();
        }

        function renderQCIssueList() {
            if (!state.qcResults) return;
            var container = document.getElementById('qcIssueList');
            container.innerHTML = '';

            var filterType = document.getElementById('qcFilterType').value;
            var grouped = groupConsecutiveIssues(state.qcResults.sortedIssues);
            var shown = 0;

            for (var ri = 0; ri < grouped.length && shown < 200; ri++) {
                var run = grouped[ri];
                if (filterType !== 'all' && run.type !== filterType) continue;

                var rep = run.representative;
                var row = document.createElement('div');
                row.className = 'qc-issue-row';

                var frameSpan = document.createElement('span');
                frameSpan.className = 'qc-issue-frame';
                if (run.startFrame === run.endFrame) {
                    frameSpan.textContent = 'F' + run.startFrame;
                } else {
                    frameSpan.textContent = 'F' + run.startFrame + '-' + run.endFrame;
                    frameSpan.title = run.frames.length + ' frames';
                }

                var typeSpan = document.createElement('span');
                typeSpan.className = 'qc-issue-type ' + rep.severity;
                typeSpan.textContent = run.type;

                var descSpan = document.createElement('span');
                descSpan.className = 'qc-issue-desc';
                descSpan.textContent = rep.description;
                if (run.frames.length > 1) descSpan.textContent += ' (' + run.frames.length + ' frames)';

                row.appendChild(frameSpan);
                row.appendChild(typeSpan);
                row.appendChild(descSpan);

                row.addEventListener('click', (function (frameIdx) {
                    return function () {
                        if (videoController) videoController.seekToFrame(frameIdx);
                    };
                })(rep.frameIdx));

                container.appendChild(row);
                shown++;
            }

            if (shown === 0) {
                container.innerHTML = '<div style="color:#888;font-size:12px;padding:8px;">No issues found</div>';
            }
        }

        function groupConsecutiveIssues(issues) {
            var sorted = issues.slice().sort(function (a, b) {
                if (a.type !== b.type) return a.type < b.type ? -1 : 1;
                if (a.identityId !== b.identityId) return (a.identityId || 0) - (b.identityId || 0);
                if (a.severity !== b.severity) return a.severity < b.severity ? -1 : 1;
                return a.frameIdx - b.frameIdx;
            });

            var severityOrder = { high: 0, medium: 1, low: 2 };
            var runs = [];
            var currentRun = null;

            for (var i = 0; i < sorted.length; i++) {
                var issue = sorted[i];
                if (currentRun &&
                    issue.type === currentRun.type &&
                    issue.identityId === currentRun.identityId &&
                    issue.severity === currentRun.severity &&
                    issue.frameIdx <= currentRun.endFrame + 2) {
                    currentRun.endFrame = issue.frameIdx;
                    currentRun.frames.push(issue.frameIdx);
                    currentRun.issues.push(issue);
                } else {
                    currentRun = {
                        type: issue.type,
                        identityId: issue.identityId,
                        severity: issue.severity,
                        startFrame: issue.frameIdx,
                        endFrame: issue.frameIdx,
                        frames: [issue.frameIdx],
                        issues: [issue],
                    };
                    runs.push(currentRun);
                }
            }

            // Pick representative from each run
            for (var j = 0; j < runs.length; j++) {
                var rIdx = Math.floor(Math.random() * runs[j].issues.length);
                runs[j].representative = runs[j].issues[rIdx];
            }

            runs.sort(function (a, b) {
                if (severityOrder[a.severity] !== severityOrder[b.severity])
                    return severityOrder[a.severity] - severityOrder[b.severity];
                return a.startFrame - b.startFrame;
            });

            return runs;
        }

        // Draggable histogram thresholds
        (function setupQCHistogramDrag() {
            var configs = [
                { canvasId: 'qcHistReproj', threshKey: 'reproj', distKey: 'reproj', title: 'Reprojection (px)' },
                { canvasId: 'qcHistEpipolar', threshKey: 'epipolar', distKey: 'epipolar', title: 'Epipolar (px)' },
            ];

            for (var ci = 0; ci < configs.length; ci++) {
                (function (cfg) {
                    var canvas = document.getElementById(cfg.canvasId);
                    if (!canvas) return;
                    var dragging = false;

                    function xToValue(e) {
                        var layout = canvas._histLayout;
                        if (!layout) return null;
                        var rect = canvas.getBoundingClientRect();
                        var scaleX = canvas.width / rect.width;
                        var px = (e.clientX - rect.left) * scaleX;
                        var frac = (px - layout.padLeft) / layout.plotWidth;
                        frac = Math.max(0.01, Math.min(1, frac));
                        return frac * layout.displayMax;
                    }

                    function isNearThreshold(e) {
                        if (!state.qcResults) return false;
                        var layout = canvas._histLayout;
                        if (!layout) return false;
                        var thresh = state.qcResults.autoThresholds[cfg.threshKey];
                        var rect = canvas.getBoundingClientRect();
                        var scaleX = canvas.width / rect.width;
                        var px = (e.clientX - rect.left) * scaleX;
                        var threshPx = layout.padLeft + (thresh / layout.displayMax) * layout.plotWidth;
                        return Math.abs(px - threshPx) < 10;
                    }

                    canvas.addEventListener('mousemove', function (e) {
                        if (dragging) {
                            var val = xToValue(e);
                            if (val != null && state.qcResults) {
                                state.qcResults.autoThresholds[cfg.threshKey] = val;
                                QC.drawHistogram(canvas, state.qcResults.distributions[cfg.distKey],
                                    val, { title: cfg.title, color: '#6b7280' });
                            }
                        } else {
                            canvas.style.cursor = isNearThreshold(e) ? 'ew-resize' : 'default';
                        }
                    });

                    canvas.addEventListener('mousedown', function (e) {
                        if (isNearThreshold(e)) {
                            dragging = true;
                            e.preventDefault();
                        }
                    });

                    window.addEventListener('mouseup', function () {
                        if (dragging) {
                            dragging = false;
                            canvas.style.cursor = 'default';
                        }
                    });
                })(configs[ci]);
            }
        })();
```
